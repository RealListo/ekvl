###DWEMER
#Dwemer Bonus
character_event = {
	id = VLRB.101
	title = VLRB.101.title
	desc = VLRB.101.desc
	
	min_age = 16
	only_capable = yes
	religion = dwemer_religion

	trigger = {
		NOT = {	has_character_modifier = dwemer_buff }		
	}

		name = "Bless Reddit"
